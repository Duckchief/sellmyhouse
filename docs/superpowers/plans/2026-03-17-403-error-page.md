# 403 Error Page Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the plain-emoji 403 block in `error.njk` with a full-screen arcade experience: 2×3 Space Invaders formation patrolling the full viewport width, matching the 401 page aesthetic.

**Architecture:** Single template edit — the `{% elif statusCode == 403 %}` block in `error.njk` is replaced with a self-contained inline `<style>` + HTML + `<script>` block identical in structure to the existing 401 block. No new files, no server changes, no tests required (pure presentational template with no business logic).

**Tech Stack:** Nunjucks, vanilla JS, HTML5 Canvas

**Spec:** `docs/superpowers/specs/2026-03-17-403-error-page-design.md`

      var INV_ROWS = 8, INV_COLS = 11;

      var PX = 10;
      var INV_W = INV_COLS * PX;
      var INV_H = INV_ROWS * PX;
      var GAP_X = 28, GAP_Y = 20;

      var FORM_W = 3 * INV_W + 2 * GAP_X;
      var FORM_H = 2 * INV_H + GAP_Y;
      var GLOW_PAD = 40;
      var BAND_H = FORM_H + GLOW_PAD * 2;

      var canvas = document.getElementById('inv-canvas-403');
      var W = window.innerWidth;
      canvas.width  = W;
      canvas.height = BAND_H;
      var ctx = canvas.getContext('2d');

      var x = (W - FORM_W) / 2;
      var y = GLOW_PAD;
      var dx = 1.8;
      var frame = 0;

      function drawRow(row, g, color) {
        ctx.shadowColor = color;
        ctx.shadowBlur  = 32;
        ctx.fillStyle   = color;
        for (var col = 0; col < 3; col++) {
          var ox = Math.round(x + col * (INV_W + GAP_X));
          var oy = Math.round(y + row * (INV_H + GAP_Y));
          for (var r = 0; r < INV_ROWS; r++) {
            for (var c = 0; c < INV_COLS; c++) {
              if (g[r][c]) ctx.fillRect(ox + c * PX, oy + r * PX, PX, PX);
            }
          }
        }
        ctx.shadowBlur = 0;
        for (var col2 = 0; col2 < 3; col2++) {
          var ox2 = Math.round(x + col2 * (INV_W + GAP_X));
          var oy2 = Math.round(y + row * (INV_H + GAP_Y));
          for (var r2 = 0; r2 < INV_ROWS; r2++) {
            for (var c2 = 0; c2 < INV_COLS; c2++) {
              if (g[r2][c2]) ctx.fillRect(ox2 + c2 * PX, oy2 + r2 * PX, PX, PX);
            }
          }
        }
      }

      function draw() {
        ctx.clearRect(0, 0, W, BAND_H);
        var g = frame === 0 ? FRAME_1 : FRAME_2;
        drawRow(0, g, '#ff66cc');
        drawRow(1, g, '#ff9933');
      }

      draw();
      setInterval(function () { frame = 1 - frame; draw(); }, 600);

      (function patrol() {

        if (x + FORM_W > W - GLOW_PAD || x < GLOW_PAD) dx = -dx;
        draw();
        requestAnimationFrame(patrol);
      })();

      window.addEventListener('resize', function () {
